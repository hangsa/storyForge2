> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the Branch Simulation feature in Phase 2 — wire the dead LLM path through the API, add missing test coverage (save/list roundtrip, prompt fallback, integration E2E, frontend), extend confidence to "high", and tighten input validation.

**Architecture:** Currently `BranchSimulator` is constructed in the API without `model_router`, so the LLM path is unreachable. We wire `get_model_router()` into both endpoints, add a `description` length cap, extend `LLMInference.confidence` to include `"high"`, mirror that in the frontend `ConfidenceBadge` and prompt YAML, remove the unused `impact_analyzer` parameter, and add tests covering every path that was previously only mocked.

**Tech Stack:** Python 3.9 + FastAPI + Pydantic dataclass + pytest-asyncio; React 18 + TypeScript + vitest + React Testing Library.

@pytest.mark.asyncio
async def test_llm_inference_uses_fallback_when_yaml_missing(simulator):
    """When branch_simulation_llm.yaml is missing, fallback prompt is used and LLM still gets called."""
    fake_prompts_dir = Path(tempfile.mkdtemp())
    (Path(__file__).parent).mkdir(parents=True, exist_ok=True)
    # Mock Path(__file__).parent.parent / "prompts" / "branch_simulation_llm.yaml" to nonexistent
    with patch("backend.conductor.branch_simulator.Path") as MockPath:
        # Make the YAML path resolve to a missing file
        def path_factory(*args, **kwargs):
            if "branch_simulation_llm.yaml" in str(args):
                return Path("/nonexistent/branch_simulation_llm.yaml")
            return Path(*args, **kwargs)
        MockPath.side_effect = path_factory
        # But we still need file open to raise FileNotFoundError
        with patch("builtins.open", side_effect=FileNotFoundError):
            result = await simulator._run_llm_inference(
                "any_project", "test description", {
                    "chapter_range": (1, 1), "characters": [], "foreshadowings": [],
                    "growth_shifts": {}, "reader_metrics": {},
                }
            )
    # Fallback should still produce a call result
    assert "tension" in result or result == {}


@pytest.mark.asyncio
async def test_llm_inference_uses_fallback_when_yaml_corrupt(simulator):
    """When YAML file is unparseable, fallback string is used."""
    with patch("builtins.open", side_effect=Exception("corrupt yaml")):
        result = await simulator._run_llm_inference(
            "any_project", "test", {
                "chapter_range": (1, 1), "characters": [], "foreshadowings": [],
                "growth_shifts": {}, "reader_metrics": {},
            }
        )
    # Either the LLM got the fallback prompt and returned valid JSON,
    # or the LLM failed gracefully — both are acceptable
    assert isinstance(result, dict)
